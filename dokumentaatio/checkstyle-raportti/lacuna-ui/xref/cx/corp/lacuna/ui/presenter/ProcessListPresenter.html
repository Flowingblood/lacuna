1   package cx.corp.lacuna.ui.presenter;
2   
3   import cx.corp.lacuna.ui.model.ProcessListModel;
4   import cx.corp.lacuna.ui.view.ProcessListView;
5   
6   import java.util.ArrayList;
7   import java.util.List;
8   
9   public class ProcessListPresenter implements Presenter, ProcessListCallbacks {
10  
11      private List<ProcessChosenEventListener> processChosenCallbacks;
12      private final ProcessListView view;
13      private final ProcessListModel model;
14  
15      public ProcessListPresenter(ProcessListView view, ProcessListModel model) {
16          if (view == null || model == null) {
17              throw new IllegalArgumentException("Arguments cannot be null!");
18          }
19          this.view = view;
20          this.model = model;
21          this.processChosenCallbacks = new ArrayList<>();
22      }
23  
24      @Override
25      public void processChosen() {
26          view.getChosenProcess()
27              .ifPresent(proc -> processChosenCallbacks.forEach(c -> c.processChosen(proc)));
28      }
29  
30      @Override
31      public void updateRequested() {
32          view.setProcessList(model.getProcesses());
33      }
34  
35      @Override
36      public void initialize() {
37          view.attach(this);
38      }
39  
40      public void addProcessChosenListener(ProcessChosenEventListener callback) {
41          if (callback == null) {
42              throw new IllegalArgumentException("Callback cannot be null!");
43          }
44          processChosenCallbacks.add(callback);
45      }
46  }
