35      private <T> ExceptionEaterProxy<T> createMemoryAccessProxy(T object) {
36          InvocationProxy<T> proxy = new PassthroughProxy<>(object);
37          ExceptionEaterProxy<T> eaterProxy = new ExceptionEaterProxy<>(
38              proxy,
39              MemoryAccessException.class,
40              ProcessOpenException.class
41          );
42  
43          // If we get a `ProcessOpenException`, we can assume that subsequent reads
44          // will fail as well so mark the current process as dead and prevent
45          // subsequent reads!
46          Consumer<Exception> processDeadListener = ex -> this.process = null;
47          eaterProxy.addEatListener(ProcessOpenException.class, processDeadListener);
48  
49          return eaterProxy;
50      }

127         throwIfOffsetOver32Bit(offset);
128         throwIfCopyingTooManyBytes(binaryData.getDataSize());
129         if (process == null) {
130             return;
131         }
132 
133         byte[] buffer = new byte[(int) binaryData.getDataSize()]; // plz forgive, no time for buffering
134         binaryData.copyToArray(0, buffer, 0, buffer.length);
135         writer.invoke(w -> w.write(process, toLacunaOffset(offset), buffer));
136     }
