1   package cx.corp.lacuna.ui.view;

3   import javax.swing.event.DocumentEvent;
4   import javax.swing.event.DocumentListener;
5   import javax.swing.text.JTextComponent;
6   import java.util.function.Consumer;
7   
8   /**
9    * Provides one-way binding for a document.
10   */
11  class UpdateDocumentListener implements DocumentListener {
12  
13      private final JTextComponent component;
14      private final Consumer<String> callback;
15  
16      private UpdateDocumentListener(JTextComponent component, Consumer<String> callback) {
17          if (callback == null || component == null) {
18              throw new IllegalArgumentException("Arguments cannot be null1");
19          }
20          this.component = component;
21          this.callback = callback;
22      }
23  
24      public static void addTo(JTextComponent component, Consumer<String> callback) {
25          DocumentListener listener = new UpdateDocumentListener(component, callback);
26          component.getDocument().addDocumentListener(listener);
27      }
28  
29      @Override
30      public void insertUpdate(DocumentEvent e) {
31          callback.accept(component.getText());
32      }
33  
34      @Override
35      public void removeUpdate(DocumentEvent e) {
36          callback.accept(component.getText());
37      }
38  
39      @Override
40      public void changedUpdate(DocumentEvent e) {
41          callback.accept(component.getText());
42      }
43  }
