1   package cx.corp.lacuna.ui;
2   
3   import cx.corp.lacuna.core.LacunaBootstrap;
4   import cx.corp.lacuna.core.ProcessOpenException;
5   import cx.corp.lacuna.ui.model.MainModel;
6   import cx.corp.lacuna.ui.model.SettingsModel;
7   import cx.corp.lacuna.ui.presenter.MainPresenter;
8   import cx.corp.lacuna.ui.view.ChooseProcessDialog;
9   import cx.corp.lacuna.ui.view.DataInspectorComponent;
10  import cx.corp.lacuna.ui.view.MainWindow;
11  import cx.corp.lacuna.ui.view.MemoryComponent;
12  
13  import javax.swing.*;
14  
15  public class LacunaUI implements Runnable {
16      public static void main(String... args) {
17          SwingUtilities.invokeLater(new LacunaUI());
18      }
19  
20      @Override
21      public void run() {
22          setSystemLookAndFeel();
23  
24          SettingsModel settings = new SettingsModel();
25          settings.setBootstrap(LacunaBootstrap.forCurrentPlatform());
26  
27          ProcessBinaryData memoryProvider = new ProcessBinaryData(
28              settings.getBootstrap().getMemoryReader(),
29              settings.getBootstrap().getMemoryWriter()
30          );
31          MemoryComponent memoryComponent = new MemoryComponent(memoryProvider);
32  
33          DataInspectorComponent dataInspector = new DataInspectorComponent(
34              settings.getBootstrap().getMemoryReader(),
35              settings.getBootstrap().getMemoryWriter()
36          );
37          memoryComponent.addCaretMovedListener((caretPosition, section) ->
38              dataInspector.setDataOffset((int) caretPosition.getDataPosition()));
39          MainModel mainModel = new MainModel();
40          mainModel.addObserver((o, arg) -> {
41              MainModel model = (MainModel) o;
42              memoryProvider.setActiveProcess(model.getActiveProcess());
43              memoryComponent.notifyProviderUpdated();
44              dataInspector.setActiveProcess(model.getActiveProcess());
45          });
46          MainWindow mainWindow = new MainWindow(new ChooseProcessDialog(settings));
47          mainWindow.setMemoryPanel(memoryComponent.getPanel());
48          mainWindow.setDataInspectorPanel(dataInspector.getPanel());
49          MainPresenter presenter = new MainPresenter(mainWindow, mainModel);
50  
51          memoryProvider.setMemoryAccessExceptionHandler(ProcessOpenException.class, ex -> {
52              // ¯\_(ツ)_/¯'
53              mainWindow.setActiveProcess(null);
54              presenter.newActiveProcessSelected();
55              mainWindow.refresh(); // validate & repaint so that the active process label updates
56              JOptionPane.showMessageDialog(
57                  memoryComponent.getPanel(),
58                  "An error occurred while opening the process!\n" +
59                      "Please check that the target process is still running and " +
60                      "that you have necessary privileges!",
61                  "Error",
62                  JOptionPane.ERROR_MESSAGE
63              );
64          });
65  
66          presenter.initialize();
67          mainWindow.show();
68      }
69  
70      private void setSystemLookAndFeel() {
71          try {
72              UIManager.setLookAndFeel(
73                  UIManager.getSystemLookAndFeelClassName());
74          } catch (Exception ex) {
75          }
76      }
77  }
