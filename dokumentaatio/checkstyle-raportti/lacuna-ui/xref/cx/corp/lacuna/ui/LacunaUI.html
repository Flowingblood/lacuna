14  public class LacunaUI implements Runnable {
15      public static void main(String... args) {
16          SwingUtilities.invokeLater(new LacunaUI());
17      }
18  
19      @Override
20      public void run() {
21          SettingsModel settings = new SettingsModel();
22          settings.setBootstrap(LacunaBootstrap.forCurrentPlatform());
23  
24          ProcessBinaryData memoryProvider = new ProcessBinaryData(
25              settings.getBootstrap().getMemoryReader(),
26              settings.getBootstrap().getMemoryWriter()
27          );
28          MemoryComponent memoryComponent = new MemoryComponent(memoryProvider);
29  
30          MainModel mainModel = new MainModel();
31          mainModel.addObserver((o, arg) -> {
32              MainModel model = (MainModel) o;
33              memoryProvider.setActiveProcess(model.getActiveProcess());
34          });
35          MainWindow mainWindow = new MainWindow(new ChooseProcessDialog(settings));
36          mainWindow.setMemoryPanel(memoryComponent.getPanel());
37          MainPresenter presenter = new MainPresenter(mainWindow, mainModel);
38          presenter.initialize();
39          mainWindow.show();
40      }
41  }
