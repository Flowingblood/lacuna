1   package cx.corp.lacuna.core.linux;
2   
3   import org.apache.commons.lang3.StringUtils;
4   import java.util.Optional;
5   
6   /**
7    * A parser for {@code /proc/:pid/cmdline} files.
8    *
9    * <p>On Linux systems, the command line that was used to launch a process can
10   * be fetched from {@code /proc/:pid/cmdline}, where {@code :pid} is the
11   * process's identifier. The file is described in {@code man proc(5)}:
12   * <p><pre>
13   * {@code
14   * /proc/[pid]/cmdline
15   *     This read-only file holds the complete command line for the process,
16   *     unless the process is a zombie.  In the latter case, there is nothing
17   *     in this file: that is, a read on this file will return 0 characters.
18   *     The command-line arguments appear in this file as a set of strings
19   *     separated by null bytes ('\0'), with a further null byte after the last
20   *     string.
21   * }
22   * </pre>
23   */
24  public class CmdlineFileParser {
25      private static final char ARG_SEPARATOR = '\0';
26      private static final char ARG_SEPARATOR_REPLACEMENT = ' ';
27  
28      /**
29       * Parses a {@code cmdline} file and returns it in space delimited form.
30       * @param cmdlineFileContents the contents of the {@code cmdline} file.
31       * @return the command line in space delimited form.
32       */
33      public Optional<String> parse(String cmdlineFileContents) {
34          return Optional.ofNullable(cmdlineFileContents)
35              .map(s -> StringUtils.strip(s, ARG_SEPARATOR + "")) // trim nulls from the end before replacing so that args ending in space don't get cut
36              .map(s -> s.replace(ARG_SEPARATOR, ARG_SEPARATOR_REPLACEMENT));
37  
38      }
39  }
