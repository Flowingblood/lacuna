1   package cx.corp.lacuna.core.windows.winapi;
2   
3   import com.sun.jna.FunctionMapper;
4   import com.sun.jna.NativeLibrary;
5   
6   import java.lang.reflect.Method;
7   
8   /**
9    * Provides mapping from camelCase Java method names to PascalCase WinAPI names.
10   */
11  public class CamelToPascalCaseFunctionMapper implements FunctionMapper {
12      @Override
13      public String getFunctionName(NativeLibrary nativeLibrary, Method method) {
14          return capitalizeFirstLetter(method.getName());
15      }
16  
17      private static String capitalizeFirstLetter(String string) {
18          if (string.length() == 1) {
19              return string.toUpperCase();
20          }
21          return string.substring(0, 1).toUpperCase() + string.substring(1);
22      }
23  }
