7   import java.io.IOException;
8   import java.nio.charset.StandardCharsets;
9   import java.nio.file.Files;
10  import java.nio.file.Path;
11  import java.util.Optional;
12  
13  
14  /**
15   * {@inheritDoc}
16   *
17   * <p>Process information gathering on Linux platforms revolves around the
18   * virtual {@code /proc} filesystem (see {@code man proc(5)}). To get the
19   * process description, the {@code /proc/:pid/cmdline} file is parsed. To
20   * get the owner, the owner of the {@code /proc/:pid} directory is fetched.
21   */
22  public class LinuxNativeProcessCollector implements NativeProcessCollector {
23  
24      private static final String PROC_CMDLINE_PATH = "cmdline";
25      private final CmdlineFileParser cmdlineParser;
26      private final Path procRoot;
27  
28      /**
29       * Constructs a new {@link LinuxNativeProcessCollector} with the specified process
30       * directory path. In most cases, this should be {@code Paths.get("/proc")}.
31       * @param procRoot the parent folder of the processes.
32       * @throws NullPointerException if {@code procRoot} is null.
33       */
34      public LinuxNativeProcessCollector(Path procRoot) {
35          if (procRoot == null) {
36              throw new NullPointerException("Proc root cannot be null!");
37          }
38          this.cmdlineParser = new CmdlineFileParser();
39          this.procRoot = procRoot;
40      }
41  
42      @Override
43      public NativeProcess collect(int pid) {
44          Path procPath = procRoot.resolve(Integer.toUnsignedString(pid));
45          NativeProcess process = readNativeProcess(procPath);
46          process.setPid(pid);
47          return process;
48      }
49  
50      private NativeProcess readNativeProcess(Path processDir) {
51          NativeProcess process = new NativeProcessImpl();
52  
53          String processCmdLine = readCmdLine(processDir);
54          process.setDescription(processCmdLine);
55  
56          String owner = readOwner(processDir);
57          process.setOwner(owner);
58  
59          return process;
60      }
61  
62      private String readCmdLine(Path processDirectory) {
63          // get process command line for description
64          // "/proc/[pid]/cmdline
65          //     This read-only file holds the complete command line for the
66          //     process, unless the process is a zombie.  In the latter case,
67          //     there is nothing in this file: that is, a read on this file
68          //     will return 0 characters.  The command-line arguments appear
69          //     in this file as a set of strings separated by null bytes
70          //     ('\0'), with a further null byte after the last string."
71          Path cmdLinePath = constructCmdLinePath(processDirectory);
72          return readContentsFully(cmdLinePath)
73              .flatMap(cmdlineParser::parse)
74              .orElse(NativeProcess.UNKNOWN_DESCRIPTION);
75      }
76  
77      private Path constructCmdLinePath(Path processDirectory) {
78          return processDirectory.resolve(PROC_CMDLINE_PATH);
79      }
80  
81      private Optional<String> readContentsFully(Path path) {
82          try {
83              String contents = new String(Files.readAllBytes(path), StandardCharsets.UTF_8);
84              return Optional.of(contents);
85          } catch (IOException ex) {
86              return Optional.empty();
87          }
88      }
89  
90      private String readOwner(Path procDirectory) {
91          try {
92              return Files.getOwner(procDirectory).getName();
93          } catch (UnsupportedOperationException | IOException e) {
94              return NativeProcess.UNKNOWN_OWNER;
95          }
96      }
97  }
