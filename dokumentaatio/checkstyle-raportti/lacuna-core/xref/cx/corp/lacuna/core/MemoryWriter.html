1   package cx.corp.lacuna.core;

3   import cx.corp.lacuna.core.domain.NativeProcess;
4   
5   /**
6    * Provides methods for writing bytes and common Java data types to the
7    * memory of a native process.
8    *
9    * @see NativeProcessEnumerator
10   * @see MemoryReader
11   */
12  public interface MemoryWriter {
13      /**
14       * Writes an 8-bit boolean value to the specified offset.
15       * True will be written as a {@code 1}, and false as a {@code 0}.
16       *
17       * @param process The native process whose memory to write to.
18       * @param offset  The memory address offset to write to.
19       * @param value   THe value to write to memory. True will write 1, false will write 0.
20       * @throws MemoryAccessException if writing fails due to, for example, an
21       *                               access violation or insufficient rights.
22       * @throws NullPointerException  if {@code process} is null.
23       */
24      void writeBoolean(NativeProcess process, int offset, boolean value) throws MemoryAccessException;
25  
26      /**
27       * Writes a byte to the specified offset.
28       *
29       * @param process The native process whose memory to write to.
30       * @param offset  The memory address offset to write to.
31       * @param value   The value to write to memory.
32       * @throws MemoryAccessException if writing fails due to, for example, an
33       *                               access violation or insufficient rights.
34       * @throws NullPointerException  if {@code process} is null.
35       */
36      void writeByte(NativeProcess process, int offset, byte value) throws MemoryAccessException;
37  
38      /**
39       * Writes one 8-bit UTF-8 code unit to the specified offset.
40       * <p>Notice that this method only writes one byte, so characters which
41       * are encoded into multiple bytes are not supported. If attempting to
42       * write characters higher than U+007F, this method will throw an exception.
43       * To write multi-byte characters, use {@link #writeStringUTF8}.
44       *
45       * @param process The native process whose memory to write to.
46       * @param offset  The memory address offset to write to.
47       * @param value   The value to write to memory.
48       * @throws MemoryAccessException    if writing fails due to, for example, an
49       *                                  access violation or insufficient rights.
50       * @throws NullPointerException     if {@code process} is null.
51       * @throws IllegalArgumentException if the character is encoded into multiple bytes.
52       * @see #writeCharUTF16LE
53       * @see #writeStringUTF8
54       * @see #writeStringUTF16LE
55       */
56      void writeCharUTF8(NativeProcess process, int offset, char value) throws MemoryAccessException;
57  
58      /**
59       * Writes one 16-bit UTF-16 little endian code unit to the specified offset.
60       * <p>Notice that this method only writes one 16-bit code unit, so characters
61       * which are encoded with two surrogate units are not supported. If attempting
62       * to write characters higher than U+FFFFF, this method will throw an exception.
63       * To write multiple code units, use {@link #writeStringUTF16LE}
64       *
65       * @param process The native process whose memory to write to.
66       * @param offset  The memory address offset to write to.
67       * @param value   The value to write to memory.
68       * @throws MemoryAccessException    if writing fails due to, for example, an
69       *                                  access violation or insufficient rights.
70       * @throws NullPointerException     if {@code process} is null.
71       * @throws IllegalArgumentException if the character is encoded into multiple bytes.
72       * @see #writeCharUTF8
73       * @see #writeStringUTF16LE
74       * @see #writeStringUTF8
75       */
76      void writeCharUTF16LE(NativeProcess process, int offset, char value) throws MemoryAccessException;
77  
78      /**
79       * Writes a little-endian 16-bit short integer to the specified offset.
80       *
81       * @param process The native process whose memory to write to.
82       * @param offset  The memory address offset to write to.
83       * @param value   The value to write to memory.
84       * @throws MemoryAccessException if writing fails due to, for example, an
85       *                               access violation or insufficient rights.
86       * @throws NullPointerException  if {@code process} is null.
87       * @cx.littleendian
88       */
89      void writeShort(NativeProcess process, int offset, short value) throws MemoryAccessException;
90  
91      /**
92       * Writes a little-endian 32-bit integer to the specified offset.
93       *
94       * @param process The native process whose memory to write to.
95       * @param offset  The memory address offset to write to.
96       * @param value   The value to write to memory.
97       * @throws MemoryAccessException if writing fails due to, for example, an
98       *                               access violation or insufficient rights.
99       * @throws NullPointerException  if {@code process} is null.
100      * @cx.littleendian
101      */
102     void writeInt(NativeProcess process, int offset, int value) throws MemoryAccessException;
103 
104     /**
105      * Writes an 32-bit IEEE 754 single precision floating point value
106      * to the specified offset.
107      *
108      * @param process The native process whose memory to write to.
109      * @param offset  The memory address offset to write to.
110      * @param value   The value to write to memory.
111      * @throws MemoryAccessException if writing fails due to, for example, an
112      *                               access violation or insufficient rights.
113      * @throws NullPointerException  if {@code process} is null.
114      * @cx.littleendian
115      */
116     void writeFloat(NativeProcess process, int offset, float value) throws MemoryAccessException;
117 
118     /**
119      * Writes a little-endian 64-bit integer to the specified offset.
120      *
121      * @param process The native process whose memory to write to.
122      * @param offset  The memory address offset to write to.
123      * @param value   The value to write to memory.
124      * @throws MemoryAccessException if writing fails due to, for example, an
125      *                               access violation or insufficient rights.
126      * @throws NullPointerException  if {@code process} is null.
127      * @cx.littleendian
128      */
129     void writeLong(NativeProcess process, int offset, long value) throws MemoryAccessException;
130 
131     /**
132      * Writes an 64-bit IEEE 754 double precision floating point value
133      * to the specified offset.
134      *
135      * @param process The native process whose memory to write to.
136      * @param offset  The memory address offset to write to.
137      * @param value   The value to write to memory.
138      * @throws MemoryAccessException if writing fails due to, for example, an
139      *                               access violation or insufficient rights.
140      * @throws NullPointerException  if {@code process} is null.
141      * @cx.littleendian
142      */
143     void writeDouble(NativeProcess process, int offset, double value) throws MemoryAccessException;
144 
145     /**
146      * Writes an UTF-8 encoded string to the specified offset.
147      * <p>Notice that UTF-8 is a variable length character encoding, in which
148      * Unicode code points higher than U+007F are encoded with two to four 8-bit
149      * code units. Users should be careful when writing characters which use multiple
150      * code units, as {@link String#length()} returns the amount of UTF-16 code units.
151      * Use {@code String.getBytes(StandardCharsets.UTF_8)} to retrieve the amount of
152      * bytes written by this method.
153      *
154      * @param process The native process whose memory to write to.
155      * @param offset  The memory address offset to write to.
156      * @param value   The value to write to memory.
157      * @throws MemoryAccessException if writing fails due to, for example, an
158      *                               access violation or insufficient rights.
159      * @throws NullPointerException  if {@code process} or {@code value} is null.
160      * @throws IllegalArgumentException if {@code value} is empty.
161      */
162     void writeStringUTF8(NativeProcess process, int offset, String value) throws MemoryAccessException;
163 
164     /**
165      * Writes an UTF-16 little-endian encoded string to the specified offset.
166      * <p>Notice that UTF-16 is a variable length character encoding, in which
167      * Unicode code points higher than U+FFFF are encoded with two 16-bit surrogate
168      * code units. Users should be careful when writing characters which use multiple
169      * code units, as {@link String#length()} returns the amount of 16-bit UTF-16 code units.
170      * Use {@code String.getBytes(StandardCharsets.UTF_16LE)} to retrieve the amount of
171      * bytes written by this method.
172      *
173      * @param process The native process whose memory to write to.
174      * @param offset  The memory address offset to write to.
175      * @param value   The value to write to memory.
176      * @throws MemoryAccessException if writing fails due to, for example, an
177      *                               access violation or insufficient rights.
178      * @throws NullPointerException  if {@code process} or {@code value} is null.
179      * @throws IllegalArgumentException if {@code value} is empty.
180      */
181     void writeStringUTF16LE(NativeProcess process, int offset, String value) throws MemoryAccessException;
182 
183     /**
184      * Writes an array of bytes to the specified offset.
185      *
186      * @param process The native process whose memory to write to.
187      * @param offset  The memory address offset to write to.
188      * @param values   The value to write to memory.
189      * @throws MemoryAccessException if writing fails due to, for example, an
190      *                               access violation or insufficient rights.
191      * @throws NullPointerException  if {@code process} or {@code values} is null.
192      * @throws IllegalArgumentException if {@code values} is empty.
193      */
194     void write(NativeProcess process, int offset, byte[] values) throws MemoryAccessException;
195 }
