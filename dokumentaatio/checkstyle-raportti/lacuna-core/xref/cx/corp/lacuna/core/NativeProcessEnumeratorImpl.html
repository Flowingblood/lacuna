1   package cx.corp.lacuna.core;
2   
3   import cx.corp.lacuna.core.domain.NativeProcess;
4   
5   import java.util.List;
6   import java.util.Objects;
7   import java.util.stream.Collectors;
8   
9   /**
10   * {@inheritDoc}
11   * <p>This implementation first fetches all processes with a
12   * {@link PidEnumerator}, then collects the processes' details with a
13   * {@link NativeProcessCollector}.
14   */
15  public class NativeProcessEnumeratorImpl implements NativeProcessEnumerator {
16      private final NativeProcessCollector processCollector;
17      private final PidEnumerator pidEnumerator;
18  
19      /**
20       * Constructs a new {@code NativeProcessEnumeratorImpl} with the specified
21       * PID enumerator and process detail collector.
22       * @param pidEnumerator the PID enumerator.
23       * @param processCollector the process detail collector.
24       * @throws NullPointerException if {@code pidEnumerator} or
25       *                              {@code processCollector} is null.
26       */
27      public NativeProcessEnumeratorImpl(PidEnumerator pidEnumerator,
28                                         NativeProcessCollector processCollector) {
29          Objects.requireNonNull(pidEnumerator, "pidEnumerator cannot be null!");
30          Objects.requireNonNull(processCollector, "processCollector cannot be null!");
31          this.pidEnumerator = pidEnumerator;
32          this.processCollector = processCollector;
33      }
34  
35      @Override
36      public List<NativeProcess> getProcesses() {
37          return pidEnumerator
38              .getPids()
39              .stream()
40              .map(processCollector::collect)
41              .collect(Collectors.toList());
42      }
43  }
