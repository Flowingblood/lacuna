1   package cx.corp.lacuna.core.linux;

3   import cx.corp.lacuna.core.ProcessOpenException;
4   
5   import java.io.IOException;
6   import java.nio.channels.SeekableByteChannel;
7   import java.nio.file.Files;
8   import java.nio.file.Path;
9   import java.nio.file.StandardOpenOption;
10  
11  /**
12   * Provides read and write access to a process's memory via the virtual
13   * {@code /proc/[pid]/mem} file.
14   */
15  public class FileMemoryProvider implements ReadableMemoryProvider, WritableMemoryProvider {
16  
17      private static final String MEM_FILE_NAME = "mem";
18      private final Path procRoot;
19  
20      /**
21       * Constructs a new {@link FileMemoryProvider} with the specified process
22       * directory path. In most cases, this should be {@code Paths.get("/proc")}.
23       * @param procRoot the parent folder of the processes.
24       * @throws NullPointerException if {@code procRoot} is null.
25       */
26      public FileMemoryProvider(Path procRoot) {
27          if (procRoot == null) {
28              throw new NullPointerException("Paths cannot be null!");
29          }
30          this.procRoot = procRoot;
31      }
32  
33      /**
34       * {@inheritDoc}
35       * <p>This method opens a channel to the virtual
36       * {@code /proc/[pid]/mem} file.
37       */
38      @Override
39      public SeekableByteChannel openRead(int pid) throws ProcessOpenException {
40          Path memFile = createPathToProcessMemoryFile(pid);
41          try {
42              return Files.newByteChannel(memFile, StandardOpenOption.READ);
43          } catch (IOException ex) {
44              String message = String.format(
45                  "Failed to open process %d for reading! See getCause()!",
46                  pid
47              );
48              throw new ProcessOpenException(message, ex);
49          }
50      }
51  
52      /**
53       * {@inheritDoc}
54       * <p>This method opens a channel to the virtual
55       * {@code /proc/[pid]/mem} file.
56       */
57      @Override
58      public SeekableByteChannel openWrite(int pid) throws ProcessOpenException {
59          Path memFile = createPathToProcessMemoryFile(pid);
60          try {
61              return Files.newByteChannel(memFile, StandardOpenOption.WRITE);
62          } catch (IOException ex) {
63              String message = String.format(
64                  "Failed to open process %d for writing! See getCause()!",
65                  pid
66              );
67              throw new ProcessOpenException(message, ex);
68          }
69      }
70  
71      private Path createPathToProcessMemoryFile(int pid) {
72          String pidStr = Integer.toString(pid);
73          return procRoot.resolve(pidStr).resolve(MEM_FILE_NAME);
74      }
75  }
