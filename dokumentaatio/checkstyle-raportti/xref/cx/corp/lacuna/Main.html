52      public static void main(String[] args) throws IOException {
53          setupPlatformSpecificStuff();
54          setupSerializers();
55  
56          List<NativeProcess> processes = processEnumerator.getProcesses();
57          for (NativeProcess proc : processes) {
58              System.out.printf(
59                      "%-5d    %-8s    %s%n",
60                      proc.getPid(),
61                      proc.getOwner(),
62                      proc.getDescription());
63          }
64  
65          Scanner scanner = new Scanner(System.in);
66          System.out.print("Enter target PID: ");
67          final int targetPid = scanner.nextInt();
68  
69          Optional<NativeProcess> proc = processes
70                  .stream()
71                  .filter(p -> p.getPid() == targetPid)
72                  .findFirst();
73  
74          if (!proc.isPresent()) {
75              System.out.println("Process " + targetPid +  " not found!");
76              return;
77          }
78  
79          System.out.print("Enter target offset: 0x");
80          int offset = scanner.nextInt(16);
81          System.out.print("How many bytes to read? 0x");
82          int count = scanner.nextInt(16);
83  
84          byte[] bytes = memoryReader.read(
85                  proc.get(),
86                  offset,
87                  count);
88  
89          for (int i = 0; i < bytes.length; i++) {
90              if (i != 0 && i % 16 == 0) {
91                  System.out.println();
92              }
93              System.out.printf("%02X ", bytes[i]);
94          }
95          System.out.println();
96  
97          if (bytes.length >= 0x10) {
98              TypeSerializer<Integer> intSerializer = typeSerializers.find(Integer.class);
99              byte[] firstInt = Arrays.copyOfRange(bytes, 0, 4);
100             byte[] secondInt = Arrays.copyOfRange(bytes, 4, 8);
101             System.out.println("firstInt = " + intSerializer.deserialize(firstInt));
102             System.out.println("secondInt = " + intSerializer.deserialize(secondInt));
103 
104 
105             TypeSerializer<Boolean> boolSerializer = typeSerializers.find(Boolean.class);
106             byte[] firstBool = Arrays.copyOfRange(bytes, 8, 9);
107             byte[] secondBool = Arrays.copyOfRange(bytes, 9, 10);
108 
109             System.out.println("firstBool = " + boolSerializer.deserialize(firstBool));
110             System.out.println("secondBool = " + boolSerializer.deserialize(secondBool));
111         }
112     }
113 
114     private static void setupSerializers() {
115         typeSerializers = new TypeSerializersImpl();
116         typeSerializers.register(Boolean.class, new Boolean8Serializer());
117         typeSerializers.register(Integer.class, new Int32Serializer());
118     }
119 
120     private static void setupPlatformSpecificStuff() {
121         if (Platform.isWindows()) {
122             setupForWindows();
123         } else {
124             setupForLinux();
125         }
126     }
127 
128     private static void setupForWindows() {
129         Map<String, Object> options = new HashMap<>();
130         // Use a mapper so that we can use Java-style function names in the interfaces
131         FunctionMapper nameMapper = new CamelToPascalCaseFunctionMapper();
132         options.put(Library.OPTION_FUNCTION_MAPPER, nameMapper);
133 
134         Kernel32 kernel = Native.loadLibrary("Kernel32", Kernel32.class, options);
135         Psapi psapi = Native.loadLibrary("Psapi", Psapi.class, options);
136         Advapi32 advapi = Native.loadLibrary("Advapi32", Advapi32.class, options);
137 
138         PidEnumerator enumerator = new WinApiPidEnumerator(psapi);
139         ProcessOpener procOpener = new WinApiProcessOpener(kernel);
140         ProcessOwnerGetter ownerGetter = new WinApiProcessOwnerGetter(advapi);
141         ProcessDescriptionGetter descriptionGetter = new WinApiProcessDescriptionGetter(kernel);
142         NativeProcessCollector collector =
143             new WindowsNativeProcessCollector(procOpener, ownerGetter, descriptionGetter);
144         processEnumerator = new WindowsNativeProcessEnumerator(enumerator, collector);
145         memoryReader = new WindowsMemoryReader(procOpener, kernel);
146     }
147 
148     private static void setupForLinux() {
149         Path procRoot = LinuxConstants.DEFAULT_PROC_ROOT;
150         processEnumerator = new LinuxNativeProcessEnumerator(procRoot);
151         FileMemoryProvider memProvider = new FileMemoryProvider(Paths.get("/proc"), "mem");
152         memoryReader = new LinuxMemoryReader(memProvider);
153     }
