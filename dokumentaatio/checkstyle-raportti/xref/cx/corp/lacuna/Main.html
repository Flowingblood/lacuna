28  public class Main {
29  
30      private static NativeProcessEnumerator processEnumerator;
31      private static MemoryReader memoryReader;
32  
33      public static void main(String[] args) throws IOException {
34          setupPlatformSpecificStuff();
35  
36          List<NativeProcess> processes = processEnumerator.getProcesses();
37          for (NativeProcess proc : processes) {
38              System.out.printf(
39                      "%-5d    %-8s    %s%n",
40                      proc.getPid(),
41                      proc.getOwner(),
42                      proc.getDescription());
43          }
44  
45          Scanner scanner = new Scanner(System.in);
46          System.out.print("Enter target PID: ");
47          final int targetPid = scanner.nextInt();
48  
49          Optional<NativeProcess> proc = processes
50                  .stream()
51                  .filter(p -> p.getPid() == targetPid)
52                  .findFirst();
53  
54          if (!proc.isPresent()) {
55              System.out.println("Process " + targetPid +  " not found!");
56              return;
57          }
58  
59          System.out.print("Enter target offset: 0x");
60          int offset = scanner.nextInt(16);
61          System.out.print("How many bytes to read? 0x");
62          int count = scanner.nextInt(16);
63  
64          byte[] bytes = memoryReader.read(
65                  proc.get(),
66                  offset,
67                  count);
68  
69          for (int i = 0; i < bytes.length; i++) {
70              if (i != 0 && (i + 1) % 16 == 0) {
71                  System.out.println();
72              }
73              System.out.printf("%02X ", bytes[i]);
74          }
75          System.out.println();
76      }
77  
78      private static void setupPlatformSpecificStuff() {
79          if (Platform.isWindows()) {
80              setupForWindows();
81          } else {
82              setupForLinux();
83          }
84      }
85  
86      private static void setupForWindows() {
87          Map<String, Object> options = new HashMap<>();
88          // Use a mapper so that we can use Java-style function names in the interfaces
89          FunctionMapper nameMapper = new CamelToPascalCaseFunctionMapper();
90          options.put(Library.OPTION_FUNCTION_MAPPER, nameMapper);
91  
92          Kernel32 kernel = Native.loadLibrary("Kernel32", Kernel32.class, options);
93          Psapi psapi = Native.loadLibrary("Psapi", Psapi.class, options);
94          Advapi32 advapi = Native.loadLibrary("Advapi32", Advapi32.class, options);
95  
96          processEnumerator = new WindowsNativeProcessEnumerator(kernel, psapi, advapi);
97          memoryReader = new WindowsMemoryReader(kernel);
98      }
99  
100     private static void setupForLinux() {
101         Path procRoot = LinuxConstants.DEFAULT_PROC_ROOT;
102         processEnumerator = new LinuxNativeProcessEnumerator(procRoot);
103         memoryReader = new LinuxMemoryReader(procRoot);
104     }
105 }
