54      public static void main(String[] args) throws IOException {
55          setupPlatformSpecificStuff();
56  
57          List<NativeProcess> processes = processEnumerator.getProcesses();
58          for (NativeProcess proc : processes) {
59              System.out.printf(
60                  "%-5d    %-8s    %s%n",
61                  proc.getPid(),
62                  proc.getOwner(),
63                  proc.getDescription());
64          }
65  
66          Scanner scanner = new Scanner(System.in);
67          System.out.print("Enter target PID: ");
68          final int targetPid = scanner.nextInt();
69  
70          Optional<NativeProcess> proc = processes
71              .stream()
72              .filter(p -> p.getPid() == targetPid)
73              .findFirst();
74  
75          if (!proc.isPresent()) {
76              System.out.println("Process " + targetPid + " not found!");
77              return;
78          }
79  
80          System.out.print("Enter target data structure offset: 0x");
81          int offset = scanner.nextInt(16);
82  
83          NativeProcess targetProcess = proc.get();
84          readAndDumpTestTargetData(targetProcess, offset);
85  
86          System.out.print("New value of int1: ");
87          int int1 = scanner.nextInt();
88          System.out.print("New value of int2: ");
89          int int2 = scanner.nextInt();
90          System.out.print("New value of bool3: ");
91          boolean bool3 = scanner.nextBoolean();
92          System.out.print("New value of bool4: ");
93          boolean bool4 = scanner.nextBoolean();
94          System.out.print("New value of short5: ");
95          short short5 = scanner.nextShort();
96          System.out.print("New value of int6: ");
97          int int6 = scanner.nextInt();
98  
99          writeTestTargetData(targetProcess, offset, int1, int2, bool3, bool4, short5, int6);
100 
101         readAndDumpTestTargetData(targetProcess, offset);
102     }

104     private static void writeTestTargetData(NativeProcess process,
105                                             int offset,
106                                             int int1,
107                                             int int2,
108                                             boolean bool3,
109                                             boolean bool4,
110                                             short short5,
111                                             int int6) {
112         memoryWriter.writeInt(process, offset + 0, int1);
113         memoryWriter.writeInt(process, offset + 4, int2);
114         memoryWriter.writeBoolean(process, offset + 8, bool3);
115         memoryWriter.writeBoolean(process, offset + 9, bool4);
116         memoryWriter.writeShort(process, offset + 10, short5);
117         memoryWriter.writeInt(process, offset + 12, int6);
118     }
119 
120     private static void readAndDumpTestTargetData(NativeProcess process, int offset) {
121         byte[] bytes = memoryReader.read(process, offset, 0x10);
122         hexDump(bytes);
123 
124         System.out.print(memoryReader.readInt(process, offset + 0) + " ");
125         System.out.print(memoryReader.readInt(process, offset + 4) + " ");
126         System.out.print(memoryReader.readBoolean(process, offset + 8) + " ");
127         System.out.print(memoryReader.readBoolean(process, offset + 9) + " ");
128         System.out.print(memoryReader.readShort(process, offset + 10) + " ");
129         int str5Ptr = memoryReader.readInt(process, offset + 12);
130         System.out.print(str5Ptr);
131         System.out.print(" (" + memoryReader.readStringUTF8(process, str5Ptr, 0x100) + ")");
132         System.out.println();
133     }

135     private static void hexDump(byte[] bytes) {
136         for (int i = 0; i < bytes.length; i++) {
137             if (i != 0 && i % 16 == 0) {
138                 System.out.println();
139             }
140             System.out.printf("%02X ", bytes[i]);
141         }
142         System.out.println();
143     }

145     private static byte[] listToBytes(List<Integer> payload) {
146         byte[] data = new byte[payload.size()];
147         for (int i = 0; i < payload.size(); i++) {
148             data[i] = payload.get(i).byteValue();
149         }
150         return data;
151     }

161     private static void setupForWindows() {
162         Map<String, Object> options = new HashMap<>();
163         // Use a mapper so that we can use Java-style function names in the interfaces
164         FunctionMapper nameMapper = new CamelToPascalCaseFunctionMapper();
165         options.put(Library.OPTION_FUNCTION_MAPPER, nameMapper);
166 
167         Kernel32 kernel = Native.loadLibrary("Kernel32", Kernel32.class, options);
168         Psapi psapi = Native.loadLibrary("Psapi", Psapi.class, options);
169         Advapi32 advapi = Native.loadLibrary("Advapi32", Advapi32.class, options);
170 
171         PidEnumerator enumerator = new WindowsPidEnumerator(psapi);
172         ProcessOpener procOpener = new WindowsProcessOpener(kernel);
173         ProcessOwnerGetter ownerGetter = new WindowsProcessOwnerGetter(advapi, kernel);
174         ProcessDescriptionGetter descriptionGetter = new WindowsProcessDescriptionGetter(kernel);
175         NativeProcessCollector collector =
176             new WindowsNativeProcessCollector(procOpener, ownerGetter, descriptionGetter);
177 
178         processEnumerator = new NativeProcessEnumeratorImpl(enumerator, collector);
179 
180         RawMemoryReader rawMemoryReader = new WindowsRawMemoryReader(procOpener, kernel);
181         memoryReader = new MemoryReaderImpl(rawMemoryReader);
182 
183         RawMemoryWriter rawWriter = new WindowsRawMemoryWriter(procOpener, kernel);
184         memoryWriter = new MemoryWriterImpl(rawWriter);
185     }
186 
187     private static void setupForLinux() {
188         Path procRoot = LinuxConstants.DEFAULT_PROC_ROOT;
189         PidEnumerator enumerator = new LinuxPidEnumerator(procRoot, LinuxConstants.DEFAULT_PID_MAX);
190         NativeProcessCollector collector = new LinuxNativeProcessCollector(procRoot);
191 
192         processEnumerator = new NativeProcessEnumeratorImpl(enumerator, collector);
193 
194         FileMemoryProvider memProvider = new FileMemoryProvider(
195             Paths.get("/proc"),
196             Paths.get("mem"));
197         RawMemoryReader rawMemoryReader = new LinuxRawMemoryReader(memProvider);
198         memoryReader = new MemoryReaderImpl(rawMemoryReader);
199     }
