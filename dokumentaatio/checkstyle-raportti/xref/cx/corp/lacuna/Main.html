37          List<NativeProcess> processes = enumerator.getProcesses();
38          for (NativeProcess proc : processes) {
39              System.out.printf(
40                      "%-5d    %-8s    %s%n",
41                      proc.getPid(),
42                      proc.getOwner(),
43                      proc.getDescription());
44          }
45  
46          MemoryReader reader = new WindowsMemoryReader(kernel);
47          NativeProcess process =
48              processes.stream()
49                  .filter(proc -> proc.getDescription().contains("TestTarget"))
50                  .findFirst()
51                  .get();
52          byte[] bytes = reader.read(process, 0x8AAC38, 16);
