45      private static NativeProcessEnumerator processEnumerator;
46      private static MemoryReader memoryReader;
47  
48      public static void main(String[] args) throws IOException {
49          setupPlatformSpecificStuff();
50  
51          List<NativeProcess> processes = processEnumerator.getProcesses();
52          for (NativeProcess proc : processes) {
53              System.out.printf(
54                      "%-5d    %-8s    %s%n",
55                      proc.getPid(),
56                      proc.getOwner(),
57                      proc.getDescription());
58          }
59  
60          Scanner scanner = new Scanner(System.in);
61          System.out.print("Enter target PID: ");
62          final int targetPid = scanner.nextInt();
63  
64          Optional<NativeProcess> proc = processes
65                  .stream()
66                  .filter(p -> p.getPid() == targetPid)
67                  .findFirst();
68  
69          if (!proc.isPresent()) {
70              System.out.println("Process " + targetPid +  " not found!");
71              return;
72          }
73  
74          System.out.print("Enter target offset: 0x");
75          int offset = scanner.nextInt(16);
76          System.out.print("How many bytes to read? 0x");
77          int count = scanner.nextInt(16);
78  
79          byte[] bytes = memoryReader.read(
80                  proc.get(),
81                  offset,
82                  count);
83  
84          for (int i = 0; i < bytes.length; i++) {
85              if (i != 0 && i % 16 == 0) {
86                  System.out.println();
87              }
88              System.out.printf("%02X ", bytes[i]);
89          }
90  
91          if (count >= 0x10) {
92              NativeProcess process = proc.get();
93              System.out.println(memoryReader.readInt(process, offset + 0));
94              System.out.println(memoryReader.readInt(process, offset + 4));
95              System.out.println(memoryReader.readBoolean(process, offset + 8));
96              System.out.println(memoryReader.readBoolean(process, offset + 9));
97              System.out.println(memoryReader.readShort(process, offset + 10));
98              System.out.println(memoryReader.readInt(process, offset + 12));
99          }
100 
101         System.out.println();
102     }
103 
104     private static void setupPlatformSpecificStuff() {
105         if (Platform.isWindows()) {
106             setupForWindows();
107         } else {
108             setupForLinux();
109         }
110     }
111 
112     private static void setupForWindows() {
113         Map<String, Object> options = new HashMap<>();
114         // Use a mapper so that we can use Java-style function names in the interfaces
115         FunctionMapper nameMapper = new CamelToPascalCaseFunctionMapper();
116         options.put(Library.OPTION_FUNCTION_MAPPER, nameMapper);
117 
118         Kernel32 kernel = Native.loadLibrary("Kernel32", Kernel32.class, options);
119         Psapi psapi = Native.loadLibrary("Psapi", Psapi.class, options);
120         Advapi32 advapi = Native.loadLibrary("Advapi32", Advapi32.class, options);
121 
122         PidEnumerator enumerator = new WindowsPidEnumerator(psapi);
123         ProcessOpener procOpener = new WindowsProcessOpener(kernel);
124         ProcessOwnerGetter ownerGetter = new WindowsProcessOwnerGetter(advapi, kernel);
125         ProcessDescriptionGetter descriptionGetter = new WindowsProcessDescriptionGetter(kernel);
126         NativeProcessCollector collector =
127             new WindowsNativeProcessCollector(procOpener, ownerGetter, descriptionGetter);
128 
129         processEnumerator = new NativeProcessEnumeratorImpl(enumerator, collector);
130 
131         RawMemoryReader rawMemoryReader = new WindowsRawMemoryReader(procOpener, kernel);
132         memoryReader = new MemoryReaderImpl(rawMemoryReader);
133     }
134 
135     private static void setupForLinux() {
136         Path procRoot = LinuxConstants.DEFAULT_PROC_ROOT;
137         PidEnumerator enumerator = new LinuxPidEnumerator(procRoot);
138         NativeProcessCollector collector = new LinuxNativeProcessCollector(procRoot);
139 
140         processEnumerator = new NativeProcessEnumeratorImpl(enumerator, collector);
141 
142         FileMemoryProvider memProvider = new FileMemoryProvider(Paths.get("/proc"), "mem");
143         RawMemoryReader rawMemoryReader = new LinuxRawMemoryReader(memProvider);
144         memoryReader = new MemoryReaderImpl(rawMemoryReader);
145     }
