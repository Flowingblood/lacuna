1   package cx.corp.lacuna.core.linux;
2   
3   import org.apache.commons.lang3.StringUtils;
4   import java.util.Optional;
5   
6   public class CmdlineFileParser {
7       private static final char ARG_SEPARATOR = '\0';
8       private static final char ARG_SEPARATOR_REPLACEMENT = ' ';
9   
10      // get process command line for description
11      // "/proc/[pid]/cmdline
12      //     This read-only file holds the complete command line for the
13      //     process, unless the process is a zombie.  In the latter case,
14      //     there is nothing in this file: that is, a read on this file
15      //     will return 0 characters.  The command-line arguments appear
16      //     in this file as a set of strings separated by null bytes
17      //     ('\0'), with a further null byte after the last string."
18  
19      public Optional<String> parse(String cmdlineFileContents) {
20          return Optional.ofNullable(cmdlineFileContents)
21              .map(s -> StringUtils.strip(s, ARG_SEPARATOR + "")) // trim nulls from the end before replacing so that args ending in space don't get cut
22              .map(s -> s.replace(ARG_SEPARATOR, ARG_SEPARATOR_REPLACEMENT));
23  
24      }
25  }
