1   package cx.corp.lacuna.core.linux;

3   import cx.corp.lacuna.core.MemoryReadException;
4   import cx.corp.lacuna.core.MemoryReader;
5   import cx.corp.lacuna.core.NativeProcess;
6   
7   import java.io.File;
8   import java.io.IOException;
9   import java.io.RandomAccessFile;
10  import java.nio.file.Path;
11  import java.nio.file.Paths;
12  import java.util.Arrays;
13  
14  public class LinuxMemoryReader implements MemoryReader {
15  
16      private static final String PROC_MEM_FILENAME = "mem";
17      private static final String FILE_MODE = "r";
18  
19      private final Path procRoot;
20  
21      public LinuxMemoryReader(Path procRoot) {
22          this.procRoot = procRoot;
23      }
24  
25      @Override
26      public byte[] read(NativeProcess process, int offset, int bytesToRead) {
27          File memFile = getProcessMemFile(process);
28  
29          byte[] bytes = new byte[bytesToRead];
30          int actuallyRead = 0;
31  
32          try (RandomAccessFile stream = new RandomAccessFile(memFile, FILE_MODE)) {
33              stream.seek(offset);
34              actuallyRead = stream.read(bytes, 0, bytesToRead);
35          } catch (IOException ex) {
36              String msg =
37                      "Reading process memory failed, see getCause()! Memory path is "
38                      + memFile + ".";
39              throw new MemoryReadException(msg, ex);
40          }
41  
42          if (actuallyRead < 1 && bytesToRead > actuallyRead) {
43              // was meant to read more bytes, but actually read nothing ??
44              String msg = "No bytes were read when expecting to read more! Memory path is "
45                      + memFile + ".";
46              throw new MemoryReadException(msg);
47          }
48  
49          if (actuallyRead < bytesToRead) {
50              // managed to read fewer bytes than expecting, trim array
51              return Arrays.copyOf(bytes, actuallyRead);
52          }
53  
54          return bytes;
55      }
56  
57      private File getProcessMemFile(NativeProcess process) {
58          Path mem = procRoot.resolve(Paths.get(process.getPid() + "", PROC_MEM_FILENAME));
59          return mem.toFile();
60      }
61  }
