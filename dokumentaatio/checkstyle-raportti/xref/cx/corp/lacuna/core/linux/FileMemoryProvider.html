1   package cx.corp.lacuna.core.linux;
2   
3   import java.io.IOException;
4   import java.nio.channels.SeekableByteChannel;
5   import java.nio.file.Files;
6   import java.nio.file.Path;
7   import java.nio.file.StandardOpenOption;
8   
9   /**
10   * Provides access to a process's memory via the virtual {@code /proc/[pid]/mem} file.
11   */
12  public class FileMemoryProvider implements ReadableMemoryProvider, WritableMemoryProvider {
13  
14      private static final String MEM_FILE_NAME = "mem";
15      private final Path procRoot;
16  
17      public FileMemoryProvider(Path procRoot) {
18          if (procRoot == null) {
19              throw new IllegalArgumentException("Paths cannot be null!");
20          }
21          this.procRoot = procRoot;
22      }
23  
24      /**
25       * Opens a stream to the virtual {@code /proc/[pid]/mem} file.
26       * @param pid The ID of the process.
27       * @throws IOException if the stream cannot be opened.
28       */
29      @Override
30      public SeekableByteChannel openRead(int pid) throws IOException {
31          Path memFile = createPathToProcessMemoryFile(pid);
32          return Files.newByteChannel(memFile, StandardOpenOption.READ);
33      }
34  
35      @Override
36      public SeekableByteChannel openWrite(int pid) throws IOException {
37          Path memFile = createPathToProcessMemoryFile(pid);
38          return Files.newByteChannel(memFile, StandardOpenOption.WRITE);
39      }
40  
41      private Path createPathToProcessMemoryFile(int pid) {
42          String pidStr = Integer.toString(pid);
43          return procRoot.resolve(pidStr).resolve(MEM_FILE_NAME);
44      }
45  }
