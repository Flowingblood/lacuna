3   import cx.corp.lacuna.core.NativeProcessEnumerator;
4   import cx.corp.lacuna.core.ProcessEnumerationException;
5   import cx.corp.lacuna.core.domain.NativeProcess;
6   import cx.corp.lacuna.core.domain.NativeProcessImpl;
7   
8   import java.io.File;
9   import java.io.FileFilter;
10  import java.io.IOException;
11  import java.nio.charset.StandardCharsets;
12  import java.nio.file.Files;
13  import java.nio.file.Path;
14  import java.nio.file.Paths;
15  import java.util.ArrayList;
16  import java.util.List;
17  import java.util.NoSuchElementException;
18  import java.util.Scanner;
19  
20  public class LinuxNativeProcessEnumerator implements NativeProcessEnumerator {
21  
22      private static final String PROC_CMDLINE_PATH = "cmdline";
23      private final Path procRoot;
24  
25      public LinuxNativeProcessEnumerator(Path procRoot) {
26          this.procRoot = procRoot;
27      }
28  
29      @Override
30      public List<NativeProcess> getProcesses() {
31          ArrayList<NativeProcess> processes = new ArrayList<>();
32  
33          FileFilter filter = new ProcFileFilter(readPidMax());
34          for (File procDirectory : readProcDirectories(filter)) {
35              NativeProcess process = readNativeProcess(procDirectory);
36              processes.add(process);
37          }
38  
39          return processes;
40      }
41  
42      private int readPidMax() {
43          try {
44              Path pidMaxPath = procRoot.resolve(LinuxConstants.PID_MAX_RELATIVE_PATH);
45              return new Scanner(pidMaxPath).nextInt();
46          } catch (IOException | NoSuchElementException ex) {
47              return LinuxConstants.FALLBACK_PID_MAX;
48          }
49      }
50  
51      private NativeProcess readNativeProcess(File procDirectory) {
52          NativeProcess process = new NativeProcessImpl();
53  
54          String directoryName = procDirectory.getName();
55          int pid = Integer.parseInt(directoryName);
56          process.setPid(pid);
57  
58          String processCmdLine = readCmdLine(directoryName);
59          process.setDescription(processCmdLine);
60  
61          String owner = readOwner(procDirectory.toPath());
62          process.setOwner(owner);
63  
64          return process;
65      }
66  
67      private File[] readProcDirectories(FileFilter filter) {
68          File[] procFiles = procRoot.toFile().listFiles(filter);
69          if (procFiles == null) {
70              throw new ProcessEnumerationException("Failed to list contents of proc root " + procRoot + "!");
71          }
72          return procFiles;
73      }
74  
75      private Path constructCmndLinePath(String pid) {
76          Path childPath = Paths.get(pid, PROC_CMDLINE_PATH);
77          return procRoot.resolve(childPath);
78      }
79  
80      private String readCmdLine(String pidString) {
81          // get process command line for description
82          // "/proc/[pid]/cmdline
83          //     This read-only file holds the complete command line for the
84          //     process, unless the process is a zombie.  In the latter case,
85          //     there is nothing in this file: that is, a read on this file
86          //     will return 0 characters.  The command-line arguments appear
87          //     in this file as a set of strings separated by null bytes
88          //     ('\0'), with a further null byte after the last string."
89          Path cmdLinePath = constructCmndLinePath(pidString);
90          try {
91              List<String> lines = Files.readAllLines(cmdLinePath, StandardCharsets.UTF_8);
92              return lines.size() > 0
93                  ? lines.get(0).replace('\0', ' ').trim()
94                  : NativeProcess.UNKNOWN_DESCRIPTION;
95          } catch (IOException e) {
96              return NativeProcess.UNKNOWN_DESCRIPTION;
97          }
98      }
99  
100     private String readOwner(Path procDirectory) {
101         try {
102             return Files.getOwner(procDirectory).getName();
103         } catch (UnsupportedOperationException | IOException | SecurityException e) {
104             return NativeProcess.UNKNOWN_OWNER;
105         }
106     }
107 }
