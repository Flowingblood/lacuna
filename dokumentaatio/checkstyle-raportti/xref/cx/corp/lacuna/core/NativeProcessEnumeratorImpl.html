1   package cx.corp.lacuna.core;

3   import cx.corp.lacuna.core.domain.NativeProcess;
4   
5   import java.util.List;
6   import java.util.stream.Collectors;
7   
8   public class NativeProcessEnumeratorImpl implements NativeProcessEnumerator {
9       private final NativeProcessCollector processCollector;
10      private final PidEnumerator pidEnumerator;
11  
12      public NativeProcessEnumeratorImpl(PidEnumerator pidEnumerator, NativeProcessCollector processCollector) {
13          if (pidEnumerator == null || processCollector == null) {
14              throw new IllegalArgumentException("Parameters cannot be null!");
15          }
16  
17          this.pidEnumerator = pidEnumerator;
18          this.processCollector = processCollector;
19      }
20  
21      @Override
22      public List<NativeProcess> getProcesses() {
23          return pidEnumerator
24              .getPids()
25              .stream()
26              .map(processCollector::collect)
27              .collect(Collectors.toList());
28      }
29  }
