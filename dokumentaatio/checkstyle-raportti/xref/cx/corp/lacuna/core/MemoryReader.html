5   /**
6    * The {@link MemoryReader} interface provides methods for reading bytes from the memory
7    * of a {@link NativeProcess} and reconstructing from them data in certain Java types.
8    *
9    * @see cx.corp.lacuna.core.NativeProcessEnumerator
10   */
11  public interface MemoryReader {
12      /**
13       * Reads an array of bytes from the memory of the specified {@link NativeProcess}.
14       * <p>
15       * <p>This method may return a byte array with fewer elements than {@code bytesToRead}.
16       * The return value will always be a non-null byte array with a length between
17       * {@code 0} and {@code bytesToRead}.
18       *
19       * @param process     The native process whose memory to read.
20       * @param offset      The memory address offset to read from.
21       * @param bytesToRead The amount of bytes to read, starting from the {@code offset}.
22       * @return The read bytes.
23       * @throws MemoryReadException if reading the memory fails.
24       */
25      byte[] read(NativeProcess process, int offset, int bytesToRead);
26  }
