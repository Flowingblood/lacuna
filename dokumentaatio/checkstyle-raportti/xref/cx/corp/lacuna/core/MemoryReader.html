5   /**
6    * The {@link MemoryReader} interface provides methods for reading bytes from the memory
7    * of a {@link NativeProcess}.
8    *
9    * @see cx.corp.lacuna.core.NativeProcessEnumerator
10   */
11  public interface MemoryReader {
12      /**
13       * Reads an 8-bit boolean value from the specified offset.
14       * Any nonzero value will be interpreted as true, whereas zero will be interpreted as false.
15       * @param process The native process whose memory to read.
16       * @param offset The memory address offset to read from.
17       * @return True if the read byte is nonzero, otherwise false.
18       */
19      boolean readBoolean(NativeProcess process, int offset); // 1-byte
20      byte readByte(NativeProcess process, int offset); // 1-byte
21      char readCharUTF8(NativeProcess process, int offset); // 1-byte -- DOES NOT CONSIDER VARIABLE LENGTH ENCODING
22      char readCharUTF16LE(NativeProcess process, int offset); // 2-byte, utf-16 -- DOES NOT CONSIDER VARIABLE LENGTH ENCODING
23      short readShort(NativeProcess process, int offset); // 2-byte
24      int readInt(NativeProcess process, int offset); // 4-byte
25      float readFloat(NativeProcess process, int offset); // 4-byte
26      long readLong(NativeProcess process, int offset); // 8-byte
27      double readDouble(NativeProcess process, int offset); // 8-byte
28      String readStringUTF8(NativeProcess process, int offset, int characterCount); // until a null is met or maxlength met -- DOES NOT CONSIDER VARIABLE LENGTH ENCODING
29      String readStringUTF16LE(NativeProcess process, int offset, int characterCount); // until a null is met or maxlength met -- DOES NOT CONSIDER VARIABLE LENGTH ENCODING
30  
31      /**
32       * Reads a region of memory of the specified {@link NativeProcess}.
33       * <p>This method may return a byte array with fewer elements than {@code bytesToRead}.
34       * The return value will always be a non-null byte array with a length between
35       * {@code 0} and {@code bytesToRead}.
36       *
37       * @param process     The native process whose memory to read.
38       * @param offset      The memory address offset to read from.
39       * @param bytesToRead The amount of bytes to read, starting from the {@code offset}.
40       * @return The read bytes.
41       * @throws MemoryReadException if reading the memory fails.
42       */
43      byte[] read(NativeProcess process, int offset, int bytesToRead);
44  }
