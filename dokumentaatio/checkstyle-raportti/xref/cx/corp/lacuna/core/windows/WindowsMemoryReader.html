3   import com.sun.jna.Memory;
4   import com.sun.jna.Native;
5   import com.sun.jna.ptr.IntByReference;
6   import cx.corp.lacuna.core.MemoryReadException;

8   import cx.corp.lacuna.core.NativeProcess;
9   import cx.corp.lacuna.core.ProcessAccessException;
10  import cx.corp.lacuna.core.windows.winapi.Kernel32;
11  import cx.corp.lacuna.core.windows.winapi.ProcessAccessFlags;
12  import cx.corp.lacuna.core.windows.winapi.SystemErrorCode;
13  import cx.corp.lacuna.core.windows.winapi.WinApiConstants;
14  
15  /** {@inheritDoc}
16   * Provides native process memory reading functionality on Windows platforms.
17   */
18  public class WindowsMemoryReader implements MemoryReader {
19  
20      private static final int FLAGS_READMEMORY =
21              ProcessAccessFlags.QUERY_INFORMATION | ProcessAccessFlags.VIRTUAL_MEMORY_READ;
22  
23      private final Kernel32 kernel;
24  
25      /** Instantiates a new {@link WindowsMemoryReader} instance using the specified
26       * Kernel32 WindowsAPI proxy.
27       *
28       * @param kernel The Kernel32 WindowsAPI proxy used for process memory reading.
29       */
30      public WindowsMemoryReader(Kernel32 kernel) {
31          this.kernel = kernel;
32      }
33  
34      /** {@inheritDoc}
35       *
36       * <p>This method implementation first attempts to open a handle for the specified {@code process}
37       * using the Windows API {@link Kernel32#openProcess(int, boolean, int)} method,
38       * then reads the specified memory segment using the
39       * {@link Kernel32#readProcessMemory(int, int, Memory, int, IntByReference)} method.
40       *
41       * <p>In order to retrieve a {@link NativeProcess} instance, the
42       * {@link cx.corp.lacuna.core.NativeProcessEnumerator} classes can be used.
43       *
44       * <blockquote><pre>{@code
45       * NativeProcessEnumerator enumerator = new WindowsNativeProcessEnumerator(...);
46       * List<NativeProcess> processes = enumerator.getProcesses();
47       * MemoryReader reader = new WindowsMemoryReader(...);
48       *
49       * try {
50       *     byte[] bytes = reader.read(processes[0], 0x1C5B100, 16);
51       *     for (byte b : bytes) {
52       *         System.out.println("%X ", b);
53       *     }
54       * } catch (ProcessAccessException | MemoryReadException ex) {
55       *     System.err.println(ex.toString());
56       *     if (ex.getCause() != null) {
57       *         System.err.println("Cause: " + ex.getCause());
58       *     }
59       * }
60       * }</pre></blockquote>
61       *
62       * @throws ProcessAccessException if the specified process could not be opened for reading.
63       * @throws MemoryReadException if reading the process memory fails.
64       */
65      @Override
66      public byte[] read(NativeProcess process, int offset, int bytesToRead)
67              throws ProcessAccessException, MemoryReadException {
68  
69          int handle = kernel.openProcess(FLAGS_READMEMORY, false, process.getPid());
70          if (handle == WinApiConstants.NULLPTR) {
71              throw new ProcessAccessException(
72                      "Process " + process + " could not be opened for reading! System error: "
73                      + Native.getLastError());
74          }
75  
76          Memory buffer = new Memory(bytesToRead);
77          IntByReference bytesRead = new IntByReference(0);
78          boolean success = kernel.readProcessMemory(
79                  handle,
80                  offset,
81                  buffer,
82                  bytesToRead,
83                  bytesRead);
84  
85          if (!success) {
86              int errorCode = Native.getLastError();
87              throw createReadExceptionFromErrorCode(errorCode);
88          }
89  
90          byte[] ret;
91          try {
92              ret = buffer.getByteArray(0, bytesToRead);
93          } catch (Error err) {
94              // e.g. "Invalid memory access" if reading outside memory bounds.
95              // the Memory class does bounds checking but you can never be sure since
96              // it ultimately calls the natively implemented Native.read method.
97              // This is a bit of a doomsday defensive programming scenario so not sure
98              // how to unit test this one.
99              throw new MemoryReadException(
100                     "An error occurred while reading memory. Use getCause() to get the cause.",
101                     err);
102         }
103 
104         return ret;
105     }
106 
107     private MemoryReadException createReadExceptionFromErrorCode(int nativeError) {
108         SystemErrorCode error = SystemErrorCode.fromId(nativeError);
109         String message;
110         if (error == null) {
111             message = "System error " + Native.getLastError() + " occurred.";
112         } else {
113             message = error.toString();
114         }
115         return new MemoryReadException(message);
116     }
117 }
