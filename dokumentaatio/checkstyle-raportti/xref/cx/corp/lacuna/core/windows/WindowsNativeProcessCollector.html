1   package cx.corp.lacuna.core.windows;
2   
3   import cx.corp.lacuna.core.domain.NativeProcess;
4   import cx.corp.lacuna.core.domain.NativeProcessImpl;
5   import cx.corp.lacuna.core.windows.NativeProcessCollector;
6   import cx.corp.lacuna.core.windows.ProcessDescriptionGetter;
7   import cx.corp.lacuna.core.windows.ProcessHandle;
8   import cx.corp.lacuna.core.windows.ProcessOpenException;
9   import cx.corp.lacuna.core.windows.ProcessOpener;
10  import cx.corp.lacuna.core.windows.ProcessOwnerGetter;
11  import cx.corp.lacuna.core.windows.winapi.ProcessAccessFlags;
12  
13  public class WindowsNativeProcessCollector implements NativeProcessCollector {
14  
15      private final ProcessOpener processOpener;
16      private final ProcessOwnerGetter ownerGetter;
17      private final ProcessDescriptionGetter descriptionGetter;
18  
19      public WindowsNativeProcessCollector(ProcessOpener processOpener,
20                                           ProcessOwnerGetter ownerGetter,
21                                           ProcessDescriptionGetter descriptionGetter) {
22          if (processOpener == null || ownerGetter == null || descriptionGetter == null) {
23              throw new IllegalArgumentException("Parameters cannot be null!");
24          }
25          this.processOpener = processOpener;
26          this.ownerGetter = ownerGetter;
27          this.descriptionGetter = descriptionGetter;
28      }
29  
30      public NativeProcess collect(int pid) {
31          NativeProcess process = new NativeProcessImpl(
32              pid,
33              NativeProcess.UNKNOWN_DESCRIPTION,
34              NativeProcess.UNKNOWN_OWNER);
35  
36          try (ProcessHandle handle = processOpener.open(pid, ProcessAccessFlags.QUERY_INFORMATION)) {
37              String description = descriptionGetter.get(handle).orElse(NativeProcess.UNKNOWN_DESCRIPTION);
38              String owner = ownerGetter.get(handle).orElse(NativeProcess.UNKNOWN_OWNER);
39              process.setDescription(description);
40              process.setOwner(owner);
41          } catch (ProcessOpenException ex) {
42              // TODO: log
43          }
44  
45          return process;
46      }
47  }
