1   package cx.corp.lacuna.core.windows;
2   
3   import com.sun.jna.Native;
4   import com.sun.jna.ptr.IntByReference;
5   import cx.corp.lacuna.core.windows.winapi.Advapi32;
6   import cx.corp.lacuna.core.windows.winapi.SystemErrorCode;
7   import cx.corp.lacuna.core.windows.winapi.WinApiConstants;
8   
9   import java.util.Optional;
10  
11  public class TokenOwnerNameFinder {
12  
13      private final Advapi32 advapi;
14  
15      public TokenOwnerNameFinder(Advapi32 advapi) {
16          if (advapi == null) {
17              throw new IllegalArgumentException("Advapi can't be null");
18          }
19          this.advapi = advapi;
20      }
21  
22      public Optional<String> getUserName(Advapi32.TokenUser user) {
23          // includes null terminator!
24          return getUsernameBufferLength(user)
25              .flatMap(bufLen -> lookupTokenUserName(user, bufLen));
26      }
27  
28      private Optional<Integer> getUsernameBufferLength(Advapi32.TokenUser user) {
29          IntByReference nameLength = new IntByReference(0);
30          IntByReference domainLength = new IntByReference(0);
31          IntByReference ignored = new IntByReference(0);
32  
33          boolean success =
34              advapi.lookupAccountSidW(
35                  WinApiConstants.NULL,
36                  user.user,
37                  null,
38                  nameLength,
39                  null,
40                  domainLength,
41                  ignored);
42  
43          return !success && callFailedBecauseBufferWasTooSmall()
44              ? Optional.of(nameLength.getValue())
45              : Optional.empty();
46      }
47  
48      private Optional<String> lookupTokenUserName(Advapi32.TokenUser user, int nameBufferLenWithNullTerminator) {
49          char[] nameBuffer = new char[nameBufferLenWithNullTerminator];
50          IntByReference nameBufNeededLen = new IntByReference(nameBuffer.length);
51          char[] domainBuffer = new char[WinApiConstants.MAX_DOMAIN_NAME_LENGTH];
52          IntByReference domainBufNeededLen = new IntByReference(domainBuffer.length);
53          IntByReference ignored = new IntByReference(0);
54  
55          boolean success =
56              advapi.lookupAccountSidW(
57                  WinApiConstants.NULL,
58                  user.user,
59                  nameBuffer,
60                  nameBufNeededLen,
61                  domainBuffer,
62                  domainBufNeededLen,
63                  ignored);
64  
65          int lengthWithoutNullTerminator = nameBufferLenWithNullTerminator - 1;
66          return success
67              ? Optional.of(new String(nameBuffer, 0, lengthWithoutNullTerminator))
68              : Optional.empty();
69      }
70  
71      private boolean callFailedBecauseBufferWasTooSmall() {
72          return Native.getLastError() == SystemErrorCode.INSUFFICIENT_BUFFER.getSystemErrorId();
73      }
74  }
