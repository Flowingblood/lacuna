5   /**
6    * Provides functionality for collecting the details of a native process into
7    * a {@link NativeProcess} model object using the provided process identifier.
8    */
9   @FunctionalInterface
10  public interface NativeProcessCollector {
11      /**
12       * Collects a {@link NativeProcess} model object from the provided
13       * process identifier. If a detail cannot be fetched, it will be set
14       * to the corresponding UNKNOWN_* constant.
15       *
16       * @param pid The process identifier.
17       * @return a model with the details of the specified process.
18       * @see NativeProcess#UNKNOWN_DESCRIPTION
19       * @see NativeProcess#UNKNOWN_OWNER
20       */
21      NativeProcess collect(int pid);
22  }
