1   package cx.corp.lacuna.core.windows;
2   
3   import com.sun.jna.Memory;
4   import com.sun.jna.Native;
5   import com.sun.jna.ptr.IntByReference;
6   import cx.corp.lacuna.core.windows.winapi.Advapi32;
7   import cx.corp.lacuna.core.windows.winapi.CloseHandle;
8   import cx.corp.lacuna.core.windows.winapi.SystemErrorCode;
9   import cx.corp.lacuna.core.windows.winapi.WinApiConstants;
10  
11  import java.util.Optional;
12  
13  public class WindowsProcessOwnerGetter implements ProcessOwnerGetter {
14  
15      private final Advapi32 advapi;
16      private final CloseHandle handleCloser;
17  
18      public WindowsProcessOwnerGetter(Advapi32 advapi, CloseHandle handleCloser) {
19          if (advapi == null || handleCloser == null) {
20              throw new IllegalArgumentException("Arguments cannot be null!");
21          }
22          this.advapi = advapi;
23          this.handleCloser = handleCloser;
24      }
25  
26      @Override
27      public Optional<String> get(ProcessHandle processHandle) {
28          if (processHandle == null) {
29              throw new IllegalArgumentException("processHandle cannot be null!");
30          }
31  
32          Optional<Integer> token = getProcessToken(processHandle.getNativeHandle());
33          try {
34              return token.flatMap(this::getTokenUser).flatMap(this::getUserName);
35          } finally {
36              token.ifPresent(handleCloser::closeHandle);
37          }
38      }
39  
40      private Optional<Integer> getProcessToken(int processHandle) {
41          IntByReference token = new IntByReference(0);
42          boolean success =
43              advapi.openProcessToken(
44                  processHandle,
45                  WinApiConstants.OPENPROCESSTOKEN_TOKEN_QUERY,
46                  token);
47          return success ? Optional.of(token.getValue()) : Optional.empty();
48      }
49  
50      private Optional<Advapi32.TokenUser> getTokenUser(int processToken) {
51          // First find out how big of a buffer we need, then get the information
52          // with a properly sized buffer.
53          return getTokenInfoBufferLength(processToken)
54              .flatMap(bufLen -> lookupTokenUser(processToken, bufLen));
55      }
56  
57      private Optional<String> getUserName(Advapi32.TokenUser user) {
58          // includes null terminator!
59          return getUsernameBufferLength(user)
60              .flatMap(bufLen -> lookupTokenUserName(user, bufLen));
61      }
62  
63      private Optional<Integer> getTokenInfoBufferLength(int processToken) {
64          IntByReference bytesNeeded = new IntByReference(0);
65          boolean success =
66              advapi.getTokenInformation(
67                  processToken,
68                  WinApiConstants.GETTOKENINFORMATION_TOKENUSER,
69                  null,
70                  0,
71                  bytesNeeded);
72  
73          return !success && callFailedBecauseBufferWasTooSmall()
74              ? Optional.of(bytesNeeded.getValue())
75              : Optional.empty();
76      }
77  
78      private Optional<Integer> getUsernameBufferLength(Advapi32.TokenUser user) {
79          IntByReference nameLength = new IntByReference(0);
80          IntByReference domainLength = new IntByReference(0);
81          IntByReference ignored = new IntByReference(0);
82  
83          boolean success =
84              advapi.lookupAccountSidW(
85                  WinApiConstants.NULL,
86                  user.user,
87                  null,
88                  nameLength,
89                  null,
90                  domainLength,
91                  ignored);
92  
93          return !success && callFailedBecauseBufferWasTooSmall()
94              ? Optional.of(nameLength.getValue())
95              : Optional.empty();
96      }
97  
98      private boolean callFailedBecauseBufferWasTooSmall() {
99          return Native.getLastError() == SystemErrorCode.INSUFFICIENT_BUFFER.getSystemErrorId();
100     }
101 
102     private Optional<Advapi32.TokenUser> lookupTokenUser(int token, int infoBufferLength) {
103         Memory memory = new Memory(infoBufferLength);
104         IntByReference bufferLen = new IntByReference(infoBufferLength);
105         boolean success =
106             advapi.getTokenInformation(
107                 token,
108                 WinApiConstants.GETTOKENINFORMATION_TOKENUSER,
109                 memory,
110                 (int) memory.size(),
111                 bufferLen);
112 
113         if (!success) {
114             return Optional.empty();
115         }
116 
117         try {
118             Advapi32.TokenUser user = new Advapi32.TokenUser(memory);
119             return Optional.of(user);
120         } catch (Error | Exception ex) {
121             return Optional.empty();
122         } finally {
123             // clean up token
124             handleCloser.closeHandle(token);
125         }
126     }
127 
128     private Optional<String> lookupTokenUserName(Advapi32.TokenUser user, int nameBufferLenWithNullTerminator) {
129         char[] nameBuffer = new char[nameBufferLenWithNullTerminator];
130         IntByReference nameBufNeededLen = new IntByReference(nameBuffer.length);
131         char[] domainBuffer = new char[WinApiConstants.MAX_DOMAIN_NAME_LENGTH];
132         IntByReference domainBufNeededLen = new IntByReference(domainBuffer.length);
133         IntByReference ignored = new IntByReference(0);
134 
135         boolean success =
136             advapi.lookupAccountSidW(
137                 WinApiConstants.NULL,
138                 user.user,
139                 nameBuffer,
140                 nameBufNeededLen,
141                 domainBuffer,
142                 domainBufNeededLen,
143                 ignored);
144 
145         int lengthWithoutNullTerminator = nameBufferLenWithNullTerminator - 1;
146         return success
147             ? Optional.of(new String(nameBuffer, 0, lengthWithoutNullTerminator))
148             : Optional.empty();
149     }
